public String convertToDecimal2(String inpNum, int base) {
        int len = inpNum.length(); (1)
        int num = 0; (1)
        int pow = 1; (1)

        for (int i = len - 1; i >= 0; i--) (2)
        {
            int valOfChar = 0; (3)
            char c = inpNum.charAt(i); (3)
            if (c >= '0' && c <= '9') (4)
            {
                valOfChar = (int) c - '0'; (5)
            }
            else
            {
                valOfChar = (int) c - 'A' + 10; (6)
            }

            if (valOfChar >= base) (7)
            {
                return "Invalid Number"; (8)
            }
            num += valOfChar * pow; (9)
            pow *= base; (9)
        }
        return String.valueOf(num); (10)
    }
